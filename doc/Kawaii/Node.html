def update dt
  if self.class.method_defined? :before_update
    before_update
  end

  if !@static
    @velocity.x += @gravity.x * dt
    @velocity.y += @gravity.y * dt
    @position.x += @velocity.x * dt
    @position.y += @velocity.y * dt
    @children.each do |child|
      child.update dt
    end
  end
   
  if self.class.method_defined? :after_update
    after_update
  end
end
