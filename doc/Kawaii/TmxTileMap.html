def to_s
	puts "Width: #{@width}"
	puts "Height: #{@height}"
	puts "Orientation: #{@orientation}"
	puts "Tile width: #{@tile_width}"
	puts "Tile height: #{@tile_height}"
	puts ""
	puts "Properties"
	properties.each do |k, v|
		puts "Key: #{k}, Value: #{v}"
	end

	puts ""
	puts "Layers: "
	@layers.each do |l|
		puts l.to_s
	end
end
